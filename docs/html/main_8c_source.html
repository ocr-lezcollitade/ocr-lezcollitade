27static size_t *get_layers(const char *layers, size_t *length)
28{
29 size_t *res = NULL;
30 size_t buffer = 0;
31 *length = 0;
32 for (size_t i = 0; 1; i++)
33 {
34 char c = layers[i];
35 if (c >= '0' && c <= '9')
36 {
37 buffer = buffer * 10 + (c - '0');
38 }
39 else if (c == ',')
40 {
41 *length = (*length) + 1;
42 res = (size_t *)realloc(res, (*length) * sizeof(size_t));
43 res[(*length) - 1] = buffer;
44 buffer = 0;
45 }
46 else if (c == '\0')
47 {
48 *length = (*length) + 1;
49 res = (size_t *)realloc(res, (*length) * sizeof(size_t));
50 res[(*length) - 1] = buffer;
51 break;
52 }
53 }
54
55 return res;
56}
