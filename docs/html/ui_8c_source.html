261int recursive_rmdir(char *path)
262{
263 DIR *d = opendir(path);
264 size_t path_len = strlen(path);
265 int r = -1;
266
267 if (d)
268 {
269 struct dirent *p;
270
271 r = 0;
272 while (!r && (p = readdir(d)))
273 {
274 int r2 = -1;
275 char *buf;
276 size_t len;
277
278 if (!strcmp(p->d_name, ".") || !strcmp(p->d_name, ".."))
279 continue;
280
281 len = path_len + strlen(p->d_name) + 2;
282 buf = malloc(len);
283
284 if (buf)
285 {
286 struct stat statbuf;
287
288 snprintf(buf, len, "%s/%s", path, p->d_name);
289 if (!stat(buf, &statbuf))
290 {
291 if (S_ISDIR(statbuf.st_mode))
292 r2 = recursive_rmdir(buf);
293 else
294 r2 = unlink(buf);
295 }
296 free(buf);
297 }
298 r = r2;
299 }
300 closedir(d);
301 }
302
303 if (!r)
304 r = rmdir(path);
305
306 return r;
307}
