152// Converts a colored pixel into grayscale.
153//
154// pixel_color: Color of the pixel to convert in the RGB format.
155// format: Format of the pixel used by the surface.
156Uint32 pixel_to_grayscale(Uint32 pixel_color, SDL_PixelFormat *format)
157{
158 Uint8 r, g, b;
159 SDL_GetRGB(pixel_color, format, &r, &g, &b);
160 Uint8 average = 0.3 * r + 0.59 * g + 0.11 * b;
161 Uint32 color = SDL_MapRGB(format, average, average, average);
162 return color;
163}
